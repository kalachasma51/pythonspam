class Account:
    def __init__(self,number):
        self.__number=number
        self.__balance=100
    def deposit(self,amount):
        self.__balance+=amount
        print("Account Balance is:",self.__balance)
    def withdraw(self,amount):
        if self.__balance-amount<100:
            print("Low balance for withdrawal")
        else:
            self.__balance-=amount
            print("Account Balance is:",self.__balance)
    def getbalance(self):
        return self.__balance
acclist=[]

            print("Account not found")
    elif choice==3:
       number=int(input("Enter the account number: "))

           print("Account not found")

import pandas as pd
import matplotlib.pyplot as plt
 
df1=pd.read_csv("semcie.csv")
 
total=df1.iloc[::,1::].sum(1)  #all rows and col without usn

maxidx=total.idxmax()
 
print("Highest marks=",maxmark)

Agrade=pd.Series()
Fgrade=pd.Series()
for i in range(1,len(df1.columns),2):    #to get 6 iterations and not 12
     code=df1.columns[i][:5]            #to get first 5 char without 'cie'
     total=df1.iloc[::,[i,i+1]].sum(1)        #i gives cie and i+1 gives see marks
   
     Scount=Acount=Fcount=0
     for mark in total:
         if mark>=90:
             Scount+=1
         elif mark>=80 and mark<90:
             Acount+=1

             Fcount+=1                     # gives no of sgrade a grade n fgrade in each subject
